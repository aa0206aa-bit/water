```javascript
function roundRect(x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x + r, y);
  ctx.lineTo(x + w - r, y);
  ctx.quadraticCurveTo(x + w, y, x + w, y + r);
  ctx.lineTo(x + w, y + h - r);
  ctx.quadraticCurveTo(x + w, y + h, x + w - r, y + h);
  ctx.lineTo(x + r, y + h);
  ctx.quadraticCurveTo(x, y + h, x, y + h - r);
  ctx.lineTo(x, y + r);
  ctx.quadraticCurveTo(x, y, x + r, y);
  ctx.closePath();
}
```

```javascript
function drawSlide() {
  const sx = SLIDE_START.x, sy = SLIDE_START.y;
  const ex = SLIDE_END.x,   ey = SLIDE_END.y;
  const sw = 22;

  const len = Math.hypot(ex - sx, ey - sy);
  const nx = -(ey - sy) / len;
  const ny =  (ex - sx) / len;

  ctx.beginPath();
  ctx.moveTo(sx + nx * sw / 2, sy + ny * sw / 2);
  ctx.lineTo(ex + nx * sw / 2, ey + ny * sw / 2);
  ctx.lineTo(ex - nx * sw / 2, ey - ny * sw / 2);
  ctx.lineTo(sx - nx * sw / 2, sy - ny * sw / 2);
  ctx.closePath();

  const g = ctx.createLinearGradient(sx, sy, ex, ey);
  g.addColorStop(0, '#d8d0c5');
  g.addColorStop(1, '#c5bdb2');
  ctx.fillStyle = g;
  ctx.fill();
  ctx.strokeStyle = C.gold;
  ctx.lineWidth = 2;
  ctx.stroke();
```

```javascript
function updateWater() {
  if (state.isPouring && state.gameState === 'PLAYING') {
    state.fillLevel = Math.min(1, state.fillLevel + FILL_RATE);
    state.waveAmp = Math.min(6, state.waveAmp + 0.5);
    if (state.fillLevel >= 1) {
      state.gameState = 'GAME_OVER';
      state.isPouring = false;
    }
  } else {
    state.waveAmp = Math.max(0, state.waveAmp - 0.15);
  }
  if (state.gameState === 'WIN') state.winTime += 1 / 60;
}

function drawWater() {
  if (state.fillLevel <= 0) return;
  const { x, y, w, h } = CUP;
  const r = 12;
  const waterTop = y + h * (1 - state.fillLevel);

  ctx.save();
  // 裁切到杯子內部
  ctx.beginPath();
  ctx.moveTo(x + 3, y);
  ctx.lineTo(x + 3, y + h - r);
  ctx.quadraticCurveTo(x + 3, y + h - 3, x + r, y + h - 3);
  ctx.lineTo(x + w - r, y + h - 3);
  ctx.quadraticCurveTo(x + w - 3, y + h - 3, x + w - 3, y + h - r);
  ctx.lineTo(x + w - 3, y);
  ctx.closePath();
  ctx.clip();

  // 正弦波水面
  ctx.beginPath();
  ctx.moveTo(x + 3, waterTop);
  for (let px = x + 3; px <= x + w - 3; px++) {
    const wy = waterTop + Math.sin((px - x) * 0.1 + state.time * 3) * state.waveAmp;
    ctx.lineTo(px, wy);
  }
  ctx.lineTo(x + w - 3, y + h);
  ctx.lineTo(x + 3, y + h);
  ctx.closePath();

  const g = ctx.createLinearGradient(x, waterTop, x, y + h);
  g.addColorStop(0, 'rgba(126,200,227,0.8)');
  g.addColorStop(1, 'rgba(80,160,205,0.92)');
  ctx.fillStyle = g;
  ctx.fill();
  ctx.restore();
}
```

```javascript
function spawnParticles() {
  for (let i = 0; i < 2; i++) {
    state.particles.push({
      x: SLIDE_END.x + (Math.random() - 0.5) * 6,
      y: SLIDE_END.y,
      vx: (Math.random() - 0.5) * 1.5,
      vy: 1 + Math.random() * 2,
      r: 4 + Math.random() * 4,
      life: 0,
      maxLife: 20 + Math.random() * 15,
    });
  }
  if (state.particles.length > 30)
    state.particles.splice(0, state.particles.length - 30);
}

function spawnSplash(x, y) {
  for (let i = 0; i < 3; i++) {
    state.splashes.push({
      x: x + (Math.random() - 0.5) * 10,
      y,
      r: 2,
      maxR: 8 + Math.random() * 6,
      life: 0,
      maxLife: 18,
    });
  }
}

function updateParticles() {
  if (state.isPouring && state.gameState === 'PLAYING') spawnParticles();

  const { x: cx, y: cy, w: cw, h: ch } = CUP;
  const waterTop = cy + ch * (1 - state.fillLevel);

  state.particles = state.particles.filter(p => {
    p.x += p.vx;
    p.y += p.vy;
    p.vy += 0.15;
    p.life++;
    if (p.y >= waterTop && p.x > cx && p.x < cx + cw) {
      spawnSplash(p.x, waterTop);
      return false;
    }
    return p.life < p.maxLife;
  });

  state.splashes = state.splashes.filter(s => {
    s.r += (s.maxR - s.r) * 0.2;
    s.life++;
    return s.life < s.maxLife;
  });
}

function drawParticles() {
  state.particles.forEach(p => {
    const a = (1 - p.life / p.maxLife) * 0.7;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(126,200,227,${a.toFixed(2)})`;
    ctx.fill();
  });
  state.splashes.forEach(s => {
    const a = (1 - s.life / s.maxLife) * 0.8;
    ctx.beginPath();
    ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(180,220,240,${a.toFixed(2)})`;
    ctx.lineWidth = 1.5;
    ctx.stroke();
  });
}
```

```javascript
function drawStar(cx, cy, outerR, innerR, points) {
  ctx.beginPath();
  for (let i = 0; i < points * 2; i++) {
    const angle = (i * Math.PI) / points - Math.PI / 2;
    const r = i % 2 === 0 ? outerR : innerR;
    const m = i === 0 ? 'moveTo' : 'lineTo';
    ctx[m](cx + r * Math.cos(angle), cy + r * Math.sin(angle));
  }
  ctx.closePath();
}

function drawFace() {
  const { x, y, w, h } = CUP;
  const cx = x + w / 2;
  const fy = y + h * 0.67;
  const es = 12;

  let expr;
  if      (state.gameState === 'WIN')       expr = 'star';
  else if (state.gameState === 'GAME_OVER') expr = 'dizzy';
  else if (state.fillLevel >= 0.85)         expr = 'panic';
  else if (state.fillLevel >= 0.6)          expr = 'nervous';
  else                                       expr = 'happy';

  // 眼睛
  if (expr === 'star') {
    ctx.fillStyle = '#f0c030';
    drawStar(cx - es, fy - 5, 6, 3, 5); ctx.fill();
    drawStar(cx + es, fy - 5, 6, 3, 5); ctx.fill();
  } else if (expr === 'dizzy') {
    ctx.strokeStyle = '#333'; ctx.lineWidth = 2.5; ctx.lineCap = 'round';
    const d = 4;
    [[cx - es, fy - 6], [cx + es, fy - 6]].forEach(([ex, ey]) => {
      ctx.beginPath();
      ctx.moveTo(ex - d, ey - d); ctx.lineTo(ex + d, ey + d);
      ctx.moveTo(ex + d, ey - d); ctx.lineTo(ex - d, ey + d);
      ctx.stroke();
    });
  } else {
    ctx.fillStyle = '#333';
    ctx.beginPath();
    ctx.arc(cx - es, fy - 6, 4, 0, Math.PI * 2);
    ctx.arc(cx + es, fy - 6, 4, 0, Math.PI * 2);
    ctx.fill();
    if (expr === 'panic') {
      ctx.strokeStyle = '#333'; ctx.lineWidth = 1.5;
      ctx.beginPath();
      ctx.arc(cx - es, fy - 6, 6, 0, Math.PI * 2);
      ctx.arc(cx + es, fy - 6, 6, 0, Math.PI * 2);
      ctx.stroke();
    }
  }

  // 嘴巴
  ctx.strokeStyle = '#444'; ctx.lineWidth = 2.5; ctx.lineCap = 'round';
  if (expr === 'happy' || expr === 'star') {
    ctx.beginPath();
    ctx.arc(cx, fy + 2, 9, 0, Math.PI);
    ctx.stroke();
    ctx.fillStyle = 'rgba(255,150,150,0.45)';
    ctx.beginPath(); ctx.ellipse(cx - 18, fy + 4, 7, 5, 0, 0, Math.PI * 2); ctx.fill();
    ctx.beginPath(); ctx.ellipse(cx + 18, fy + 4, 7, 5, 0, 0, Math.PI * 2); ctx.fill();
  } else if (expr === 'nervous') {
    ctx.beginPath();
    ctx.moveTo(cx - 9, fy + 8);
    ctx.quadraticCurveTo(cx - 4, fy + 5, cx, fy + 8);
    ctx.quadraticCurveTo(cx + 4, fy + 11, cx + 9, fy + 8);
    ctx.stroke();
  } else if (expr === 'panic') {
    ctx.beginPath();
    ctx.arc(cx, fy + 9, 7, 0, Math.PI * 2);
    ctx.stroke();
  } else if (expr === 'dizzy') {
    ctx.beginPath();
    ctx.arc(cx, fy + 14, 9, Math.PI, 0);
    ctx.stroke();
  }
}
```

```javascript
function resetGame() {
  state.fillLevel = 0;
  state.gameState = 'READY';
  state.isPouring = false;
  state.particles = [];
  state.splashes  = [];
  state.waveAmp   = 0;
  state.winTime   = 0;
}

function checkWinLose() {
  if (state.gameState !== 'PLAYING') return;
  if (Math.abs(state.fillLevel - TARGET_LEVEL) <= TOLERANCE)
    state.gameState = 'WIN';
  // GAME_OVER 由 updateWater() 在 fillLevel >= 1 時觸發
}

function getLogicalCoords(e) {
  const rect = canvas.getBoundingClientRect();
  const src = e.touches ? e.touches[0] : e;
  return {
    x: (src.clientX - rect.left) * (LW / rect.width),
    y: (src.clientY - rect.top)  * (LH / rect.height),
  };
}

function isInButton(lx, ly) {
  const bx = LW / 2 - 80, by = LH / 2 + 50;
  return lx >= bx && lx <= bx + 160 && ly >= by && ly <= by + 48;
}

function onPointerDown(e) {
  e.preventDefault();
  const { x: lx, y: ly } = getLogicalCoords(e);
  if (state.gameState === 'WIN' || state.gameState === 'GAME_OVER') {
    if (isInButton(lx, ly)) resetGame();
    return;
  }
  if (state.gameState === 'READY')   state.gameState = 'PLAYING';
  if (state.gameState === 'PLAYING') state.isPouring = true;
}

function onPointerUp(e) {
  e.preventDefault();
  if (!state.isPouring) return;
  state.isPouring = false;
  checkWinLose();
}

canvas.addEventListener('mousedown',   onPointerDown);
canvas.addEventListener('mouseup',     onPointerUp);
canvas.addEventListener('mouseleave',  onPointerUp);
canvas.addEventListener('touchstart',  onPointerDown, { passive: false });
canvas.addEventListener('touchend',    onPointerUp,   { passive: false });
canvas.addEventListener('touchcancel', onPointerUp,   { passive: false });
```
