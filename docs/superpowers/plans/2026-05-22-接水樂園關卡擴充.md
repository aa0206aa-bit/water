替換為：

```javascript
function cupPath(cup) {
  const { x, y, w, h } = CUP;
  const cx = x + w / 2;
  const r = 8;
  const N = 32;
  ctx.beginPath();
  for (let i = 0; i <= N; i++) {
    const t = 1 - i / N;
    const halfW = cup.widthAt(t) / 2;
    const py = y + (i / N) * (h - r);
    if (i === 0) ctx.moveTo(cx - halfW, py);
    else ctx.lineTo(cx - halfW, py);
  }
  const bHW = cup.widthAt(0) / 2;
  ctx.quadraticCurveTo(cx - bHW, y + h, cx - bHW + r, y + h);
  ctx.lineTo(cx + bHW - r, y + h);
  ctx.quadraticCurveTo(cx + bHW, y + h, cx + bHW, y + h - r);
  for (let i = N; i >= 0; i--) {
    const t = 1 - i / N;
    const halfW = cup.widthAt(t) / 2;
    const py = y + (i / N) * (h - r);
    ctx.lineTo(cx + halfW, py);
  }
  ctx.closePath();
}
```

```javascript
function drawWater(cup) {
  if (state.fillLevel <= 0) return;
  const { x, y, w, h } = CUP;
  const cx = x + w / 2;
  const waterTopY = y + h * (1 - state.fillLevel);
  const halfW = cup.widthAt(state.fillLevel) / 2;

  ctx.save();
  cupPath(cup);
  ctx.clip();

  ctx.beginPath();
  ctx.moveTo(cx - halfW, waterTopY);
  for (let px = cx - halfW; px <= cx + halfW; px++) {
    const wy = waterTopY + Math.sin((px - cx) * 0.1 + state.time * 3) * state.waveAmp;
    ctx.lineTo(px, wy);
  }
  ctx.lineTo(cx + halfW, y + h);
  ctx.lineTo(cx - halfW, y + h);
  ctx.closePath();

  const g = ctx.createLinearGradient(x, waterTopY, x, y + h);
  g.addColorStop(0, '#ff9800');
  g.addColorStop(1, '#e65100');
  ctx.fillStyle = g;
  ctx.fill();

  ctx.strokeStyle = 'rgba(255,220,100,0.8)';
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.moveTo(cx - halfW, waterTopY);
  for (let px = cx - halfW; px <= cx + halfW; px++) {
    const wy = waterTopY + Math.sin((px - cx) * 0.1 + state.time * 3) * state.waveAmp;
```

```javascript
function drawCupMini(cup, cx, topY, h, scale) {
  const r = 5;
  const N = 24;
  ctx.beginPath();
  for (let i = 0; i <= N; i++) {
    const t = 1 - i / N;
    const halfW = (cup.widthAt(t) / 2) * scale;
    const py = topY + (i / N) * (h - r);
    if (i === 0) ctx.moveTo(cx - halfW, py);
    else ctx.lineTo(cx - halfW, py);
  }
  const bHW = (cup.widthAt(0) / 2) * scale;
  ctx.quadraticCurveTo(cx - bHW, topY + h, cx - bHW + r, topY + h);
  ctx.lineTo(cx + bHW - r, topY + h);
  ctx.quadraticCurveTo(cx + bHW, topY + h, cx + bHW, topY + h - r);
  for (let i = N; i >= 0; i--) {
    const t = 1 - i / N;
    const halfW = (cup.widthAt(t) / 2) * scale;
    const py = topY + (i / N) * (h - r);
    ctx.lineTo(cx + halfW, py);

}
```

```javascript
const MINI_SCALE = 70 / 120;
```

```javascript
function onPointerDown(e) {
  e.preventDefault();
  const { x: lx, y: ly } = getLogicalCoords(e);

  if (state.gameState === 'SELECT') {
    for (let i = 0; i < LEVELS.length; i++) {
      if (i <= state.maxUnlocked) {
        const { x, y, w, h } = tileRect(i);
        if (lx >= x && lx <= x + w && ly >= y && ly <= y + h) {
          state.currentLevel = i;
          resetLevel();
        }
      }
    }
    return;
  }

  if (state.gameState === 'WIN') {
    if (isInRect(lx, ly, BTN_L)) {
      state.currentLevel < LEVELS.length - 1 ? nextLevel() : resetGame();
    } else if (isInRect(lx, ly, BTN_R)) {
      state.gameState = 'SELECT';
    }
    return;
  }

  if (state.gameState === 'GAME_OVER') {
    if (isInRect(lx, ly, BTN_L)) resetLevel();
    else if (isInRect(lx, ly, BTN_R)) state.gameState = 'SELECT';
    return;
  }

  if (state.gameState === 'READY')   state.gameState = 'PLAYING';
  if (state.gameState === 'PLAYING') state.isPouring = true;
}
```
